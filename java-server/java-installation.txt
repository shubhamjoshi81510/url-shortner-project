sudo apt-get update

sudo apt-get upgrade -y


curl https://download.oracle.com/java/20/latest/jdk-20_linux-x64_bin.deb -o jdk20.deb


sudo apt install ~/jdk20.deb -y


sudo update-alternatives --install /usr/bin/java java /usr/lib/jvm/jdk-20/bin/java 1

sudo update-alternatives --install /usr/bin/javac javac /usr/lib/jvm/jdk-20/bin/javac 1

sudo update-alternatives --install /usr/bin/jar jar /usr/lib/jvm/jdk-20/bin/jar 1

### Verify if Java Installation is successful
java --version
